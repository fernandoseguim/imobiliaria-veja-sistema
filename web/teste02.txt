		function E13706_removerDiacriticos(s) {
			var diacritics = [
					[/[\300-\306]/g, 'A'],
					[/[\340-\346]/g, 'a'],
					[/[\310-\313]/g, 'E'],
					[/[\350-\353]/g, 'e'],

					[/[\321]/g, 'N'],
					[/[\361]/g, 'n'],
					[/[\307]/g, 'C'],
					[/[\347]/g, 'c']
		    ];

			for (var i = 0; i < diacritics.length; i++) {

			}
